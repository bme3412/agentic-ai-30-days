Example 1: E2B Sandbox Basics 
from e2b_code_interpreter import Sandbox

# Create an E2B sandbox (isolated Linux VM)
with Sandbox() as sandbox:
    # Execute Python code safely
    execution = sandbox.run_code("""
import math

# This runs in complete isolation
primes = [n for n in range(2, 100)
          if all(n % i != 0 for i in range(2, int(math.sqrt(n)) + 1))]
print(f"Primes under 100: {primes}")
print(f"Count: {len(primes)}")
""")

    # Access results
    if execution.error:
        print(f"Error: {execution.error.traceback}")
    else:
        for line in execution.logs.stdout:
            print(line)

# Sandbox automatically cleaned up
Key Concepts:
  • Sandbox spins up in ~300ms
  • Context manager ensures cleanup
  • Error info includes full traceback
  • State persists between run_code calls
Example 2: smolagents CodeAgent 
from smolagents import CodeAgent, InferenceClientModel, tool

# Define a custom tool
@tool
def fetch_stock_price(symbol: str) -> float:
    """Fetches current stock price for a symbol.

    Args:
        symbol: Stock ticker symbol (e.g., 'AAPL')
    Returns:
        Current price as a float
    """
    # Mock implementation
    prices = {"AAPL": 175.50, "GOOGL": 140.25, "MSFT": 378.90}
    return prices.get(symbol, 0.0)

# Create CodeAgent with E2B sandboxing
with CodeAgent(
    model=InferenceClientModel(),
    tools=[fetch_stock_price],
    executor_type="e2b"  # Routes execution to E2B
) as agent:
    # Agent writes Python to solve the task
    result = agent.run("""
        Get the stock prices for AAPL and GOOGL.
        Calculate which one has grown more if AAPL started at $150
        and GOOGL started at $120. Show percentage growth for each.
    """)
    print(result)
CodeAgent Benefits:
  • Writes Python instead of JSON tool calls
  • Better composability with loops and conditions
  • One-line E2B integration via executor_type
  • Custom tools defined with @tool decorator
Example 3: Iteration with Error Recovery 
from e2b_code_interpreter import Sandbox
from openai import OpenAI

client = OpenAI()

def run_coding_agent(task: str, max_iterations: int = 3) -> str:
    """Run a coding agent with error recovery."""

    with Sandbox() as sandbox:
        messages = [
            {"role": "system", "content": """You are a Python expert.
Write code to solve the task. Output ONLY executable code.
If you see an error, fix it and try again."""},
            {"role": "user", "content": task}
        ]

        for i in range(max_iterations):
            # Generate code
            response = client.chat.completions.create(
                model="gpt-4o",
                messages=messages
            )
            code = response.choices[0].message.content

            # Strip markdown fences if present
            if "```python" in code:
                code = code.split("```python")[1].split("```")[0]

            print(f"--- Iteration {i+1} ---")
            print(f"Code: {code[:100]}...")

            # Execute in sandbox
            execution = sandbox.run_code(code)

            if execution.error:
                error_msg = execution.error.traceback
                print(f"Error: {error_msg[:100]}...")

                # Feed error back for refinement
                messages.append({"role": "assistant", "content": code})
                messages.append({"role": "user", "content": f"Error:\n{error_msg}\n\nFix it."})
            else:
                # Success!
                output = "\n".join(str(log) for log in execution.logs.stdout)
                print(f"Success: {output}")
                return output

        return "Max iterations reached"

# Usage
result = run_coding_agent("Calculate factorial of 20 using recursion")
Iteration Pattern:
  • Errors become observations, not failures
  • Traceback fed back to LLM for analysis
  • Set max_iterations to prevent infinite loops
  • Track token usage across iterations
Example 4: Production-Ready Agent 
from e2b_code_interpreter import Sandbox
from openai import OpenAI
from dataclasses import dataclass
import re
import logging

logger = logging.getLogger(__name__)

DANGEROUS_PATTERNS = [
    r"os\.system", r"subprocess", r"__import__",
    r"eval\(", r"exec\(", r"open\(.*['\"]w"
]

@dataclass
class ExecutionResult:
    success: bool
    output: str
    iterations: int
    error: str | None = None

class ProductionCodingAgent:
    def __init__(self, max_iterations: int = 5, timeout: int = 30):
        self.max_iterations = max_iterations
        self.timeout = timeout
        self.client = OpenAI()

    def _check_dangerous(self, code: str) -> str | None:
        """Check for dangerous patterns in generated code."""
        for pattern in DANGEROUS_PATTERNS:
            if re.search(pattern, code):
                return f"Blocked: {pattern}"
        return None

    def run(self, task: str) -> ExecutionResult:
        """Execute a task with full safety checks."""

        with Sandbox(timeout=120) as sandbox:
            messages = [
                {"role": "system", "content": "Write Python code. Output only code."},
                {"role": "user", "content": task}
            ]

            for i in range(self.max_iterations):
                response = self.client.chat.completions.create(
                    model="gpt-4o", messages=messages, temperature=0
                )
                code = response.choices[0].message.content.strip()

                # Safety check
                if danger := self._check_dangerous(code):
                    logger.warning(f"Dangerous code blocked: {danger}")
                    return ExecutionResult(False, "", i+1, danger)

                # Execute with timeout
                execution = sandbox.run_code(code, timeout=self.timeout)

                if execution.error:
                    messages.append({"role": "assistant", "content": code})
                    messages.append({"role": "user", "content": f"Error:\n{execution.error.traceback}\nFix it."})
                else:
                    output = "\n".join(str(log) for log in execution.logs.stdout)
                    return ExecutionResult(True, output, i+1)

            return ExecutionResult(False, "", self.max_iterations, "Max iterations")

# Usage
agent = ProductionCodingAgent(max_iterations=3, timeout=30)
result = agent.run("Calculate the sum of squares from 1 to 100")
print(f"Success: {result.success}, Output: {result.output}")
Production Features:
  • Dangerous pattern detection before execution
  • Structured result with iteration tracking
  • Configurable timeouts and iteration limits
  • Logging for debugging and monitoring
