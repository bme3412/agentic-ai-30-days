Basic Document Parsing

                                
                            

                            
Parse a PDF to markdown and semantic chunks with a few lines of code

                            
"""LandingAI ADE: Basic Document Parsing"""

from pathlib import Path
from landingai_ade import LandingAIADE

# Initialize client (uses VISION_AGENT_API_KEY env var)
client = LandingAIADE()

# Parse a local PDF
response = client.parse(
    document=Path("invoice.pdf"),
    model="dpt-2-latest"
)

# Access markdown output (great for RAG)
print("=== Markdown Output ===")
print(response.markdown)

# Access structured chunks
print("\n=== Document Chunks ===")
for chunk in response.chunks:
    print(f"Type: {chunk.type}")
    print(f"Text: {chunk.text[:100]}...")
    print(f"Page: {chunk.grounding.bounding_box.page}")
    print("---")

                        


                        

                            

                                
Schema-Based Field Extraction

                                
                            

                            
Extract specific fields using Pydantic schema with confidence scores

                            
"""Schema-Driven Document Extraction with Pydantic"""

from pathlib import Path
from landingai_ade import LandingAIADE
from landingai_ade.lib import pydantic_to_json_schema
from pydantic import BaseModel, Field

# Define extraction schema
class Invoice(BaseModel):
    vendor: str = Field(description="Company that issued the invoice")
    invoice_number: str = Field(description="Invoice ID or number")
    invoice_date: str = Field(description="Date invoice was issued")
    line_items: list[dict] = Field(description="List of items")
    subtotal: float = Field(description="Subtotal before tax")
    tax: float = Field(description="Tax amount")
    total: float = Field(description="Total amount due")

# Initialize client
client = LandingAIADE()

# Parse document first
parsed = client.parse(
    document=Path("invoice.pdf"),
    model="dpt-2-latest"
)

# Extract structured fields from parsed content
extraction = client.extract(
    schema=pydantic_to_json_schema(Invoice),
    markdown=parsed.markdown
)

# Access extracted data
print(f"Vendor: {extraction.data.vendor}")
print(f"Invoice #: {extraction.data.invoice_number}")
print(f"Total: ${extraction.data.total:,.2f}")

# Check extraction confidence per field
print("\n=== Confidence Scores ===")
for field, meta in extraction.metadata.items():
    status = "OK" if meta.confidence > 0.85 else "REVIEW"
    print(f"  {field}: {meta.confidence:.2f} [{status}]")

                        


                        

                            

                                
Visual Grounding & Verification

                                
                            

                            
Use bounding boxes to create verifiable extractions with citations

                            
"""Visual Grounding: Traceable Document Extraction"""

from pathlib import Path
from landingai_ade import LandingAIADE
import fitz  # PyMuPDF for PDF rendering

client = LandingAIADE()

# Parse with grounding information
response = client.parse(
    document=Path("financial_report.pdf"),
    model="dpt-2-latest"
)

# Find table chunks
table_chunks = [c for c in response.chunks if c.type == "table"]

for i, table in enumerate(table_chunks):
    print(f"Table {i+1}:")
    print(f"  Text: {table.text[:80]}...")

    # Access bounding box (normalized 0-1 coordinates)
    bbox = table.grounding.bounding_box
    print(f"  Page: {bbox.page}")
    print(f"  Coords: ({bbox.x1:.3f}, {bbox.y1:.3f}) to ({bbox.x2:.3f}, {bbox.y2:.3f})")


def highlight_extraction(pdf_path: str, chunk, output_path: str):
    """Render PDF page with highlighted extraction region."""
    doc = fitz.open(pdf_path)
    page = doc[chunk.grounding.bounding_box.page - 1]

    # Convert normalized coords to page coords
    rect = page.rect
    bbox = chunk.grounding.bounding_box
    highlight_rect = fitz.Rect(
        bbox.x1 * rect.width,
        bbox.y1 * rect.height,
        bbox.x2 * rect.width,
        bbox.y2 * rect.height
    )

    # Add highlight annotation
    page.add_highlight_annot(highlight_rect)

    # Save to image
    pix = page.get_pixmap()
    pix.save(output_path)
    print(f"Saved highlighted page to {output_path}")

# Highlight first table for verification
if table_chunks:
    highlight_extraction("financial_report.pdf", table_chunks[0], "table_highlight.png")

                        


                        

                            

                                
Production Pipeline with Confidence Routing

                                
                            

                            
Full production pipeline with batch processing, error handling, and human-in-the-loop

                            
"""Production Agentic Document Pipeline"""

from pathlib import Path
from dataclasses import dataclass
from landingai_ade import LandingAIADE
from landingai_ade.lib import pydantic_to_json_schema
from pydantic import BaseModel, Field
import logging
import json

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# Extraction schema
class InvoiceData(BaseModel):
    vendor: str = Field(description="Vendor company name")
    invoice_number: str = Field(description="Invoice ID")
    total: float = Field(description="Total amount due")

@dataclass
class ExtractionResult:
    file: str
    success: bool
    data: dict | None
    confidence: float
    needs_review: bool
    error: str | None = None

class InvoiceProcessor:
    def __init__(self, confidence_threshold: float = 0.85):
        self.client = LandingAIADE()
        self.schema = pydantic_to_json_schema(InvoiceData)
        self.threshold = confidence_threshold

    def process_invoice(self, file_path: Path) -> ExtractionResult:
        """Process single invoice with error handling."""
        try:
            # Step 1: Parse document
            logger.info(f"Parsing {file_path.name}...")
            parsed = self.client.parse(document=file_path, model="dpt-2-latest")

            if not parsed.markdown:
                return ExtractionResult(
                    file=str(file_path), success=False, data=None,
                    confidence=0.0, needs_review=True, error="Empty document"
                )

            # Step 2: Extract structured data
            extraction = self.client.extract(
                schema=self.schema, markdown=parsed.markdown
            )

            # Step 3: Calculate overall confidence
            confidences = [m.confidence for m in extraction.metadata.values()]
            avg_conf = sum(confidences) / len(confidences) if confidences else 0

            # Step 4: Route based on confidence
            needs_review = avg_conf < self.threshold
            if needs_review:
                logger.warning(f"{file_path.name}: Low confidence ({avg_conf:.2f})")

            return ExtractionResult(
                file=str(file_path), success=True,
                data=extraction.data.model_dump(),
                confidence=avg_conf, needs_review=needs_review
            )

        except Exception as e:
            logger.error(f"Error: {file_path.name}: {e}")
            return ExtractionResult(
                file=str(file_path), success=False, data=None,
                confidence=0.0, needs_review=True, error=str(e)
            )

    def process_batch(self, directory: Path) -> list[ExtractionResult]:
        """Process all PDFs in directory."""
        results = []
        for pdf in directory.glob("*.pdf"):
            results.append(self.process_invoice(pdf))
        return results

# Usage
processor = InvoiceProcessor(confidence_threshold=0.85)
results = processor.process_batch(Path("./invoices"))

# Summary
auto = [r for r in results if r.success and not r.needs_review]
review = [r for r in results if r.needs_review]
failed = [r for r in results if not r.success]

print(f"\n=== Summary ===")
print(f"Auto-processed: {len(auto)}")
print(f"Needs review: {len(review)}")
print(f"Failed: {len(failed)}")
