Pipeline Stages Explained

🔊

Voice Activity Detection

Distinguishes speech from background noise. Determines when the user starts and stops speaking. Modern VAD like Silero achieves 10-50ms latency.

Latency 10-50ms
📝

Speech-to-Text

Converts audio to text in real-time with streaming partial results. Deepgram Nova-3 provides best-in-class accuracy with low latency.

Latency ~200ms
🧠

Language Model

Processes transcribed text and generates response tokens. Streaming output enables TTS to start before the full response is ready.

Latency 300-800ms
🗣

Text-to-Speech

Synthesizes natural-sounding audio from text. Cartesia Sonic accepts streaming input for minimal time-to-first-audio.

Latency 100-200ms

Streaming vs Non-Streaming Latency

Why streaming at every stage is crucial for natural conversation

Non-Streaming (Sequential)

VAD 50ms
STT (wait for complete) 800ms
LLM (wait for complete) 600ms
TTS (wait for complete) 300ms
Total latency: 1750ms

Streaming (Overlapped)

VAD 50ms
STT (stream) 200ms
LLM (stream) 400ms
TTS (stream) 250ms
First Audio: 350ms
Time to first audio: 350ms
💡

5x Faster Response

Streaming reduces perceived latency from 1750ms to 350ms by overlapping pipeline stages. The user hears the first word of the response while the LLM is still generating the rest.

Conversation Latency Targets

Slow >800ms

Code Examples

Production-ready patterns for LiveKit voice agents

Basic Voice Agent

Minimal setup with VoicePipelineAgent

"""Basic Voice Agent with LiveKit Agents Framework"""

from livekit.agents import AgentSession, Agent, AgentServer
from livekit.plugins import deepgram, openai, cartesia, silero

class VoiceAssistant(Agent):
    def __init__(self):
        super().__init__(
            instructions="""You are a helpful voice assistant.
            Keep responses concise - aim for 1-2 sentences.
            Be conversational and natural."""
        )

server = AgentServer()

@server.rtc_session(agent_name="assistant")
async def entrypoint(ctx):
    session = AgentSession(
        stt=deepgram.STT(model="nova-3"),
        llm=openai.LLM(model="gpt-4.1-mini"),
        tts=cartesia.TTS(voice="79a125e8-cd45-4c13-8a67-188112f4dd22"),
        vad=silero.VAD.load(),
    )
    await session.start(room=ctx.room, agent=VoiceAssistant())

if __name__ == "__main__":
    server.run()

Voice Agent with Tools

Add function calling capabilities

"""Voice Agent with Tool Calling"""

from livekit.agents import AgentSession, Agent, AgentServer, function_tool
from livekit.plugins import deepgram, openai, cartesia, silero

@function_tool
async def get_weather(location: str) -> str:
    """Get current weather for a location."""
    # In production, call a real weather API
    return f"The weather in {location} is 72F and sunny."

@function_tool
async def set_reminder(message: str, minutes: int) -> str:
    """Set a reminder for the user."""
    return f"Reminder set: '{message}' in {minutes} minutes."

class WeatherAssistant(Agent):
    def __init__(self):
        super().__init__(
            instructions="""You help with weather and reminders.
            Keep responses brief and conversational.""",
            tools=[get_weather, set_reminder],
        )

async def entrypoint(ctx):
    session = AgentSession(
        stt=deepgram.STT(model="nova-3"),
        llm=openai.LLM(model="gpt-4.1-mini"),
        tts=cartesia.TTS(voice="79a125e8-cd45-4c13-8a67-188112f4dd22"),
        vad=silero.VAD.load(),
    )
    await session.start(room=ctx.room, agent=WeatherAssistant())

if __name__ == "__main__":
    server.run()

OpenAI Realtime (MultimodalAgent)

Audio-to-audio with lowest latency

"""Voice Agent using OpenAI Realtime API"""

from livekit.agents import AgentSession, Agent, AgentServer
from livekit.agents.multimodal import MultimodalAgent
from livekit.plugins import openai

class RealtimeAssistant(Agent):
    def __init__(self):
        super().__init__(
            instructions="""You are a friendly conversational assistant.
            Speak naturally and expressively."""
        )

server = AgentServer()

@server.rtc_session(agent_name="realtime-assistant")
async def entrypoint(ctx):
    model = openai.realtime.RealtimeModel(
        model="gpt-4o-realtime-preview",
        voice="alloy",
        temperature=0.8,
        turn_detection=openai.realtime.ServerVadOptions(
            threshold=0.5,
            prefix_padding_ms=300,
            silence_duration_ms=500,
        ),
    )

    agent = MultimodalAgent(model=model)
    session = AgentSession()

    await session.start(
        room=ctx.room,
        agent=RealtimeAssistant(),
        multimodal=agent
    )

if __name__ == "__main__":
    server.run()

Production Configuration

Error handling, fallbacks, and observability

"""Production Voice Agent with Error Handling"""

import logging
from livekit.agents import AgentSession, Agent, AgentServer
from livekit.plugins import deepgram, openai, cartesia, silero
from contextlib import asynccontextmanager

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger("voice-agent")

class ProductionAssistant(Agent):
    def __init__(self):
        super().__init__(
            instructions="""You are a professional customer service agent.
            Be helpful, concise, and empathetic."""
        )

server = AgentServer()

@asynccontextmanager
async def create_session_with_fallback():
    """Create session with fallback providers."""
    try:
        session = AgentSession(
            stt=deepgram.STT(model="nova-3", interim_results=True),
            llm=openai.LLM(model="gpt-4.1-mini", temperature=0.7),
            tts=cartesia.TTS(voice="79a125e8-cd45-4c13-8a67-188112f4dd22"),
            vad=silero.VAD.load(),
        )
        logger.info("Primary configuration loaded")
        yield session
    except Exception as e:
        logger.warning(f"Primary failed: {e}, using fallback")
        session = AgentSession(
            stt=openai.STT(model="whisper-1"),
            llm=openai.LLM(model="gpt-4.1-mini"),
            tts=openai.TTS(model="tts-1", voice="nova"),
            vad=silero.VAD.load(),
        )
        yield session

@server.rtc_session(agent_name="production-assistant")
async def entrypoint(ctx):
    logger.info(f"New session: {ctx.room.name}")

    async with create_session_with_fallback() as session:
        @session.on("user_speech_committed")
        def on_user_speech(text: str):
            logger.info(f"User: {text[:100]}...")

        @session.on("agent_speech_committed")
        def on_agent_speech(text: str):
            logger.info(f"Agent: {text[:100]}...")

        await session.start(room=ctx.room, agent=ProductionAssistant())

if __name__ == "__main__":
    server.run()
