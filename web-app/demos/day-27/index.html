4 Add YoY Comparison ↻ 
WITH current_period AS (
    SELECT DATE_TRUNC('month', o.order_date) AS month,
           p.category,
           SUM(oi.quantity * oi.unit_price) AS revenue
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    WHERE o.order_date >= DATEADD(month, -6, CURRENT_DATE)
    GROUP BY 1, 2
),
prior_year AS (
    SELECT DATE_TRUNC('month', DATEADD(year, 1, o.order_date)) AS month,
           p.category,
           SUM(oi.quantity * oi.unit_price) AS revenue_ly
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    WHERE o.order_date >= DATEADD(month, -18, CURRENT_DATE)
      AND o.order_date < DATEADD(month, -12, CURRENT_DATE)
    GROUP BY 1, 2
)
SELECT
    c.month,
    c.category,
    c.revenue,
    p.revenue_ly,
    ROUND((c.revenue - p.revenue_ly) / p.revenue_ly * 100, 1) AS yoy_growth_pct
FROM current_period c
LEFT JOIN prior_year p ON c.month = p.month AND c.category = p.category
ORDER BY c.month DESC, c.revenue DESC;

import snowflake.connector

    account="your_account",

)

Query with Validation Loop

Self-correcting SQL generation with error handling

"""Text-to-SQL with Validation and Self-Correction"""

from dataclasses import dataclass
from snowflake.cortex import Analyst
import snowflake.connector
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

@dataclass
class QueryResult:
    success: bool
    sql: str
    data: list | None
    attempts: int
    error: str | None = None

class ValidatingDataAgent:
    def __init__(self, connection, semantic_model: str, max_attempts: int = 3):
        self.conn = connection
        self.analyst = Analyst(connection=connection, semantic_model=semantic_model)
        self.max_attempts = max_attempts

    def ask(self, question: str) -> QueryResult:
        """Generate and validate SQL with automatic error correction."""
        attempts = 0
        last_error = None

        while attempts < self.max_attempts:
            attempts += 1
            logger.info(f"Attempt {attempts}: Generating SQL for '{question}'")

            try:
                # Generate SQL (include error context for retries)
                if last_error:
                    enhanced_question = f"{question}\n\nPrevious error: {last_error}"
                    result = self.analyst.ask(enhanced_question)
                else:
                    result = self.analyst.ask(question)

                # Validate by executing
                logger.info(f"Executing SQL:\n{result.sql}")
                cursor = self.conn.cursor()
                cursor.execute(result.sql)
                data = cursor.fetchall()

                # Check for reasonable results
                if self._validate_results(data, question):
                    logger.info(f"Success after {attempts} attempt(s)")
                    return QueryResult(
                        success=True,
                        sql=result.sql,
                        data=data,
                        attempts=attempts
                    )
                else:
                    last_error = "Query returned unexpected results (empty or malformed)"

            except Exception as e:
                last_error = str(e)
                logger.warning(f"Attempt {attempts} failed: {last_error}")

        # Max attempts reached
        return QueryResult(
            success=False,
            sql=result.sql if result else "",
            data=None,
            attempts=attempts,
            error=last_error
        )

    def _validate_results(self, data: list, question: str) -> bool:
        """Basic result validation."""
        # Check not empty (unless question implies it could be)
        if not data and "if any" not in question.lower():
            return False
        return True

# Usage
conn = snowflake.connector.connect(...)
agent = ValidatingDataAgent(conn, "sales_semantic_model", max_attempts=3)

result = agent.ask("Show top 10 customers by revenue this year")

if result.success:
    print(f"Query succeeded in {result.attempts} attempt(s)")
    print(f"SQL: {result.sql}")
    for row in result.data:
        print(row)
else:
    print(f"Query failed after {result.attempts} attempts: {result.error}")

Full Agentic Data Pipeline

Production-grade data agent with multi-step reasoning

"""Agentic Data Pipeline with Multi-Step Reasoning"""

from anthropic import Anthropic
import snowflake.connector
from dataclasses import dataclass
from typing import Literal
import json

@dataclass
class AnalysisStep:
    step_type: Literal["clarify", "query", "analyze", "visualize"]
    description: str
    result: dict | None = None

class AgenticDataAnalyst:
    """Multi-step data analysis agent with planning and execution."""

    SYSTEM_PROMPT = """You are a data analyst agent. Given a business question:
1. Break it into logical analysis steps
2. Generate SQL for each data need
3. Interpret results in business context
4. Suggest follow-up analyses

Available tables: customers, orders, order_items, products, regions
Always validate assumptions and explain your reasoning."""

    def __init__(self, anthropic_client: Anthropic, snowflake_conn):
        self.client = anthropic_client
        self.conn = snowflake_conn
        self.conversation = []

    def analyze(self, question: str) -> dict:
        """Run full analysis pipeline for a business question."""

        # Step 1: Plan the analysis
        plan = self._plan_analysis(question)

        # Step 2: Execute each step
        results = []
        for step in plan:
            if step.step_type == "query":
                step.result = self._execute_query_step(step)
            elif step.step_type == "analyze":
                step.result = self._interpret_results(step, results)
            results.append(step)

        # Step 3: Generate final summary
        summary = self._generate_summary(question, results)

        return {
            "question": question,
            "steps": results,
            "summary": summary,
            "follow_up_questions": self._suggest_follow_ups(question, results)
        }

    def _plan_analysis(self, question: str) -> list[AnalysisStep]:
        """Use LLM to plan analysis steps."""
        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            system=self.SYSTEM_PROMPT,
            messages=[{
                "role": "user",
                "content": f"Plan analysis steps for: {question}\nReturn JSON array of steps."
            }]
        )

        steps_json = json.loads(response.content[0].text)
        return [AnalysisStep(**s) for s in steps_json]

    def _execute_query_step(self, step: AnalysisStep) -> dict:
        """Generate and execute SQL for a query step."""
        # Generate SQL using LLM
        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            system=self.SYSTEM_PROMPT,
            messages=[{
                "role": "user",
                "content": f"Generate SQL for: {step.description}\nReturn only SQL."
            }]
        )

        sql = response.content[0].text.strip()

        # Execute with retry on error
        for attempt in range(3):
            try:
                cursor = self.conn.cursor()
                cursor.execute(sql)
                data = cursor.fetchall()
                columns = [desc[0] for desc in cursor.description]
                return {"sql": sql, "columns": columns, "data": data}
            except Exception as e:
                if attempt == 2:
                    return {"sql": sql, "error": str(e)}
                # Get corrected SQL
                sql = self._fix_sql(sql, str(e))

    def _fix_sql(self, sql: str, error: str) -> str:
        """Use LLM to fix SQL errors."""
        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            messages=[{
                "role": "user",
                "content": f"Fix this SQL:\n{sql}\n\nError: {error}\n\nReturn only corrected SQL."
            }]
        )
        return response.content[0].text.strip()

    def _interpret_results(self, step: AnalysisStep, prior_results: list) -> dict:
        """Generate business interpretation of results."""
        context = json.dumps([s.result for s in prior_results if s.result])

        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            messages=[{
                "role": "user",
                "content": f"Interpret these results:\n{context}\n\nFor: {step.description}"
            }]
        )
        return {"interpretation": response.content[0].text}

    def _generate_summary(self, question: str, results: list) -> str:
        """Generate executive summary of analysis."""
        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            messages=[{
                "role": "user",
                "content": f"Summarize analysis for '{question}':\n{json.dumps([s.result for s in results])}"
            }]
        )
        return response.content[0].text

    def _suggest_follow_ups(self, question: str, results: list) -> list[str]:
        """Suggest follow-up analyses."""
        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=512,
            messages=[{
                "role": "user",
                "content": f"Suggest 3 follow-up questions after analyzing '{question}'"
            }]
        )
        return response.content[0].text.split("\n")

# Usage
client = Anthropic()
conn = snowflake.connector.connect(...)

analyst = AgenticDataAnalyst(client, conn)
analysis = analyst.analyze("Why did revenue drop 15% in Q3?")

print(f"Summary: {analysis['summary']}")
print(f"\nFollow-up questions:")
for q in analysis['follow_up_questions']:
    print(f"  - {q}")
