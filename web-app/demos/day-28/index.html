LangSmith: Zero-Config Tracing 
import os
from langchain_anthropic import ChatAnthropic
from langchain.agents import AgentExecutor, create_tool_calling_agent
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.tools import tool

# Enable LangSmith tracing with environment variables
os.environ["LANGCHAIN_TRACING_V2"] = "true"
os.environ["LANGCHAIN_API_KEY"] = "ls_..."
os.environ["LANGCHAIN_PROJECT"] = "my-agent"

@tool
def search_docs(query: str) -> str:
    """Search documentation."""
    return f"Found results for: {query}"

# Create agent - automatically traced
llm = ChatAnthropic(model="claude-sonnet-4-20250514")
prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    ("human", "{input}"),
    ("placeholder", "{agent_scratchpad}")
])

agent = create_tool_calling_agent(llm, [search_docs], prompt)
executor = AgentExecutor(agent=agent, tools=[search_docs])

# Every run is traced to LangSmith
result = executor.invoke({"input": "Find docs about authentication"})

# View at: https://smith.langchain.com
Phoenix: OpenTelemetry Tracing 
import phoenix as px
from phoenix.otel import register
from openinference.instrumentation.langchain import LangChainInstrumentor
from openinference.instrumentation.openai import OpenAIInstrumentor

# Start local Phoenix server
px.launch_app()

# Register tracer
tracer_provider = register(
    project_name="my-agent",
    endpoint="http://localhost:6006/v1/traces"
)

# Instrument frameworks
LangChainInstrumentor().instrument(tracer_provider=tracer_provider)
OpenAIInstrumentor().instrument(tracer_provider=tracer_provider)

# Now any LangChain or OpenAI calls are traced
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage

llm = ChatOpenAI(model="gpt-4o")
response = llm.invoke([HumanMessage(content="Hello!")])

# View traces at: http://localhost:6006
Custom Cost Tracking Callback 
import time
from dataclasses import dataclass, field
from langchain.callbacks.base import BaseCallbackHandler

MODEL_PRICING = {
    "claude-sonnet-4-20250514": {"input": 3.00, "output": 15.00},
    "gpt-4o": {"input": 2.50, "output": 10.00},
}

@dataclass
class TraceMetrics:
    total_cost: float = 0.0
    total_tokens: int = 0
    spans: list = field(default_factory=list)

class CostTracker(BaseCallbackHandler):
    def __init__(self):
        self.metrics = TraceMetrics()
        self._starts = {}

    def on_llm_start(self, serialized, prompts, **kwargs):
        self._starts[str(kwargs.get("run_id"))] = time.time()

    def on_llm_end(self, response, **kwargs):
        run_id = str(kwargs.get("run_id"))
        latency = (time.time() - self._starts.pop(run_id, 0)) * 1000

        usage = response.llm_output.get("usage", {})
        tokens_in = usage.get("input_tokens", 0)
        tokens_out = usage.get("output_tokens", 0)

        model = response.llm_output.get("model_name", "unknown")
        pricing = MODEL_PRICING.get(model, {"input": 0, "output": 0})
        cost = (tokens_in * pricing["input"] +
                tokens_out * pricing["output"]) / 1_000_000

        self.metrics.total_cost += cost
        self.metrics.total_tokens += tokens_in + tokens_out
        self.metrics.spans.append({
            "model": model,
            "tokens": tokens_in + tokens_out,
            "cost": cost,
            "latency_ms": latency
        })
Production Alerting 
from datetime import datetime, timedelta
from langsmith import Client

class ProductionMonitor:
    def __init__(self, project_name: str):
        self.client = Client()
        self.project = project_name
        self.thresholds = {
            "max_cost": 0.50,      # Per trace
            "max_latency_p95": 5000,  # ms
            "max_error_rate": 5.0     # percent
        }

    def check_alerts(self) -> list[dict]:
        runs = list(self.client.list_runs(
            project_name=self.project,
            start_time=datetime.utcnow() - timedelta(hours=1),
            is_root=True
        ))

        alerts = []
        errors = sum(1 for r in runs if r.error)
        error_rate = (errors / len(runs)) * 100 if runs else 0

        if error_rate > self.thresholds["max_error_rate"]:
            alerts.append({
                "severity": "critical",
                "type": "error_rate",
                "message": f"Error rate: {error_rate:.1f}%"
            })

        return alerts

# Usage
monitor = ProductionMonitor("production-agent")
alerts = monitor.check_alerts()
for alert in alerts:
    print(f"[{alert['severity']}] {alert['message']}")
