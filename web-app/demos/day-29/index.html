DeepEval: LLM Testing Framework 
"""DeepEval: LLM Testing Framework

Run with: deepeval test run test_agent.py
"""

from deepeval import assert_test
from deepeval.test_case import LLMTestCase
from deepeval.metrics import (
    AnswerRelevancyMetric,
    FaithfulnessMetric,
    HallucinationMetric,
    ContextualPrecisionMetric
)

# ══════════════════════════════════════════════════════════════
# Define test cases
# ══════════════════════════════════════════════════════════════

test_case = LLMTestCase(
    input="What is the refund policy for digital products?",
    actual_output="Digital products can be refunded within 14 days if not used.",
    expected_output="Refunds available within 14 days for unused digital products.",
    context=[
        "Digital products are refundable within 14 days of purchase.",
        "Products that have been downloaded or activated cannot be refunded.",
        "Physical products have a 30-day return window."
    ]
)

# ══════════════════════════════════════════════════════════════
# Configure metrics with thresholds
# ══════════════════════════════════════════════════════════════

faithfulness = FaithfulnessMetric(
    threshold=0.7,
    model="gpt-4o",  # Judge model
    include_reason=True
)

relevancy = AnswerRelevancyMetric(
    threshold=0.7,
    model="gpt-4o"
)

hallucination = HallucinationMetric(
    threshold=0.5,  # Lower is better for hallucination
    model="gpt-4o"
)

# ══════════════════════════════════════════════════════════════
# Run evaluation
# ══════════════════════════════════════════════════════════════

def test_refund_policy():
    """Test that agent correctly answers refund questions."""
    assert_test(
        test_case,
        metrics=[faithfulness, relevancy, hallucination]
    )

# Run all tests
if __name__ == "__main__":
    # Evaluate single test case
    faithfulness.measure(test_case)
    print(f"Faithfulness Score: {faithfulness.score}")
    print(f"Reason: {faithfulness.reason}")

    relevancy.measure(test_case)
    print(f"Relevancy Score: {relevancy.score}")

    hallucination.measure(test_case)
    print(f"Hallucination Score: {hallucination.score}")
Ragas: RAG Pipeline Evaluation 
"""Ragas: RAG Pipeline Evaluation

Evaluate both retrieval and generation quality in RAG systems.
"""

from ragas import evaluate
from ragas.metrics import (
    context_precision,
    context_recall,
    faithfulness,
    answer_relevancy
)
from datasets import Dataset

# ══════════════════════════════════════════════════════════════
# Prepare evaluation dataset
# ══════════════════════════════════════════════════════════════

eval_data = {
    "question": [
        "What is the capital of France?",
        "How does photosynthesis work?",
        "What are the symptoms of diabetes?"
    ],
    "answer": [
        "The capital of France is Paris.",
        "Photosynthesis converts sunlight into chemical energy in plants.",
        "Common symptoms include increased thirst, frequent urination, and fatigue."
    ],
    "contexts": [
        ["Paris is the capital and largest city of France."],
        ["Photosynthesis is the process by which plants convert light energy into chemical energy."],
        ["Diabetes symptoms include excessive thirst, frequent urination, unexplained weight loss, and fatigue."]
    ],
    "ground_truth": [
        "Paris is the capital of France.",
        "Photosynthesis is how plants convert sunlight to energy.",
        "Diabetes symptoms: thirst, urination, weight loss, fatigue."
    ]
}

dataset = Dataset.from_dict(eval_data)

# ══════════════════════════════════════════════════════════════
# Run Ragas evaluation
# ══════════════════════════════════════════════════════════════

results = evaluate(
    dataset,
    metrics=[
        context_precision,   # Are retrieved docs relevant?
        context_recall,      # Were all relevant docs retrieved?
        faithfulness,        # Is answer grounded in context?
        answer_relevancy     # Does answer address the question?
    ]
)

# ══════════════════════════════════════════════════════════════
# Analyze results
# ══════════════════════════════════════════════════════════════

print("Overall Scores:")
print(f"  Context Precision: {results['context_precision']:.3f}")
print(f"  Context Recall:    {results['context_recall']:.3f}")
print(f"  Faithfulness:      {results['faithfulness']:.3f}")
print(f"  Answer Relevancy:  {results['answer_relevancy']:.3f}")

# Get per-question breakdown
df = results.to_pandas()
print("\nPer-Question Results:")
print(df[['question', 'faithfulness', 'answer_relevancy']].to_string())

# Identify weak spots
low_faith = df[df['faithfulness'] < 0.7]
if len(low_faith) > 0:
    print(f"\nWarning: {len(low_faith)} questions have low faithfulness")
    print("Consider improving retrieval or adjusting generation prompts.")
Custom Agent Metrics 
"""Custom Evaluation Metrics for Agents

Build domain-specific metrics for tool usage, task completion, and efficiency.
"""

from deepeval.metrics import BaseMetric
from deepeval.test_case import LLMTestCase
from dataclasses import dataclass
from typing import Optional
import json

# ══════════════════════════════════════════════════════════════
# Tool Usage Accuracy Metric
# ══════════════════════════════════════════════════════════════

@dataclass
class ToolCall:
    name: str
    arguments: dict

class ToolAccuracyMetric(BaseMetric):
    """Evaluate if the agent called the right tool with correct arguments."""

    def __init__(
        self,
        expected_tools: list[ToolCall],
        threshold: float = 0.8
    ):
        self.expected_tools = expected_tools
        self.threshold = threshold
        self.score: Optional[float] = None
        self.reason: Optional[str] = None

    def measure(self, test_case: LLMTestCase) -> float:
        # Parse actual tool calls from output
        actual_calls = self._extract_tool_calls(test_case.actual_output)

        if not self.expected_tools:
            self.score = 1.0 if not actual_calls else 0.0
            return self.score

        # Calculate accuracy
        correct = 0
        total = len(self.expected_tools)

        for expected in self.expected_tools:
            for actual in actual_calls:
                if self._tools_match(expected, actual):
                    correct += 1
                    break

        self.score = correct / total
        self.reason = f"Matched {correct}/{total} expected tool calls"
        return self.score

    def _extract_tool_calls(self, output: str) -> list[ToolCall]:
        """Extract tool calls from agent output."""
        # Implementation depends on your agent's output format
        calls = []
        try:
            data = json.loads(output)
            if "tool_calls" in data:
                for tc in data["tool_calls"]:
                    calls.append(ToolCall(tc["name"], tc["arguments"]))
        except:
            pass
        return calls

    def _tools_match(self, expected: ToolCall, actual: ToolCall) -> bool:
        if expected.name != actual.name:
            return False
        # Check key arguments match
        for key, value in expected.arguments.items():
            if actual.arguments.get(key) != value:
                return False
        return True

    def is_successful(self) -> bool:
        return self.score >= self.threshold


# ══════════════════════════════════════════════════════════════
# Task Completion Metric
# ══════════════════════════════════════════════════════════════

class TaskCompletionMetric(BaseMetric):
    """Evaluate if the agent completed the assigned task."""

    def __init__(self, completion_criteria: list[str], threshold: float = 0.8):
        self.criteria = completion_criteria
        self.threshold = threshold
        self.score: Optional[float] = None

    def measure(self, test_case: LLMTestCase) -> float:
        output = test_case.actual_output.lower()
        met = sum(1 for c in self.criteria if c.lower() in output)
        self.score = met / len(self.criteria)
        return self.score

    def is_successful(self) -> bool:
        return self.score >= self.threshold


# ══════════════════════════════════════════════════════════════
# Efficiency Metric (tokens + steps)
# ══════════════════════════════════════════════════════════════

class EfficiencyMetric(BaseMetric):
    """Evaluate agent efficiency based on tokens used and steps taken."""

    def __init__(
        self,
        max_tokens: int = 2000,
        max_steps: int = 5,
        threshold: float = 0.7
    ):
        self.max_tokens = max_tokens
        self.max_steps = max_steps
        self.threshold = threshold
        self.score: Optional[float] = None

    def measure(self, test_case: LLMTestCase, tokens_used: int, steps: int) -> float:
        token_score = max(0, 1 - (tokens_used / self.max_tokens))
        step_score = max(0, 1 - (steps / self.max_steps))
        self.score = (token_score + step_score) / 2
        return self.score

    def is_successful(self) -> bool:
        return self.score >= self.threshold


# ══════════════════════════════════════════════════════════════
# Usage
# ══════════════════════════════════════════════════════════════

# Define expected behavior
expected_tools = [
    ToolCall("search_products", {"query": "laptop"}),
    ToolCall("check_inventory", {"product_id": "LAPTOP-001"})
]

tool_metric = ToolAccuracyMetric(expected_tools, threshold=0.8)
task_metric = TaskCompletionMetric(
    ["found product", "checked inventory", "provided recommendation"],
    threshold=0.7
)
CI/CD Regression Testing 
"""CI/CD Regression Testing for Agents

Automated testing pipeline with baseline comparison and statistical analysis.
"""

import json
import statistics
from dataclasses import dataclass
from pathlib import Path
from datetime import datetime
from deepeval import evaluate
from deepeval.test_case import LLMTestCase
from deepeval.metrics import FaithfulnessMetric, AnswerRelevancyMetric

# ══════════════════════════════════════════════════════════════
# Configuration
# ══════════════════════════════════════════════════════════════

@dataclass
class RegressionConfig:
    baseline_path: Path = Path("baselines/latest.json")
    test_suite_path: Path = Path("tests/eval_suite.json")
    output_path: Path = Path("results/")
    significance_threshold: float = 0.05  # 5% degradation allowed
    min_pass_rate: float = 0.90  # 90% of tests must pass

config = RegressionConfig()

# ══════════════════════════════════════════════════════════════
# Load test suite and baseline
# ══════════════════════════════════════════════════════════════

def load_test_suite(path: Path) -> list[LLMTestCase]:
    """Load test cases from JSON file."""
    with open(path) as f:
        data = json.load(f)

    return [
        LLMTestCase(
            input=tc["input"],
            actual_output=tc["actual_output"],
            expected_output=tc.get("expected_output"),
            context=tc.get("context", [])
        )
        for tc in data["test_cases"]
    ]

def load_baseline(path: Path) -> dict:
    """Load baseline metrics from previous run."""
    if not path.exists():
        return {"metrics": {}, "timestamp": None}
    with open(path) as f:
        return json.load(f)

# ══════════════════════════════════════════════════════════════
# Run evaluation
# ══════════════════════════════════════════════════════════════

def run_regression_tests(test_cases: list[LLMTestCase]) -> dict:
    """Run all test cases and collect metrics."""
    metrics = [
        FaithfulnessMetric(threshold=0.7),
        AnswerRelevancyMetric(threshold=0.7)
    ]

    results = {
        "faithfulness": [],
        "relevancy": [],
        "passed": 0,
        "failed": 0,
        "failures": []
    }

    for i, tc in enumerate(test_cases):
        try:
            eval_result = evaluate([tc], metrics)
            f_score = eval_result[0].metrics[0].score
            r_score = eval_result[0].metrics[1].score

            results["faithfulness"].append(f_score)
            results["relevancy"].append(r_score)

            if f_score >= 0.7 and r_score >= 0.7:
                results["passed"] += 1
            else:
                results["failed"] += 1
                results["failures"].append({
                    "index": i,
                    "input": tc.input[:100],
                    "faithfulness": f_score,
                    "relevancy": r_score
                })
        except Exception as e:
            results["failed"] += 1
            results["failures"].append({"index": i, "error": str(e)})

    return results

# ══════════════════════════════════════════════════════════════
# Compare with baseline
# ══════════════════════════════════════════════════════════════

def compare_with_baseline(current: dict, baseline: dict) -> dict:
    """Statistical comparison with baseline."""
    comparison = {"regressions": [], "improvements": [], "status": "PASS"}

    for metric in ["faithfulness", "relevancy"]:
        if metric not in baseline.get("metrics", {}):
            continue

        curr_mean = statistics.mean(current[metric])
        base_mean = baseline["metrics"][metric]["mean"]
        diff = (curr_mean - base_mean) / base_mean

        if diff < -config.significance_threshold:
            comparison["regressions"].append({
                "metric": metric,
                "baseline": base_mean,
                "current": curr_mean,
                "change": f"{diff*100:.1f}%"
            })
            comparison["status"] = "FAIL"
        elif diff > config.significance_threshold:
            comparison["improvements"].append({
                "metric": metric,
                "change": f"+{diff*100:.1f}%"
            })

    # Check pass rate
    total = current["passed"] + current["failed"]
    pass_rate = current["passed"] / total if total > 0 else 0
    if pass_rate < config.min_pass_rate:
        comparison["status"] = "FAIL"
        comparison["regressions"].append({
            "metric": "pass_rate",
            "baseline": config.min_pass_rate,
            "current": pass_rate
        })

    return comparison

# ══════════════════════════════════════════════════════════════
# Main pipeline
# ══════════════════════════════════════════════════════════════

def main():
    print("Loading test suite...")
    test_cases = load_test_suite(config.test_suite_path)
    baseline = load_baseline(config.baseline_path)

    print(f"Running {len(test_cases)} test cases...")
    results = run_regression_tests(test_cases)

    print("Comparing with baseline...")
    comparison = compare_with_baseline(results, baseline)

    # Output results
    print(f"\n{'='*50}")
    print(f"REGRESSION TEST: {comparison['status']}")
    print(f"{'='*50}")
    print(f"Passed: {results['passed']}/{results['passed']+results['failed']}")

    if comparison["regressions"]:
        print("\nRegressions detected:")
        for r in comparison["regressions"]:
            print(f"  - {r['metric']}: {r.get('change', 'below threshold')}")

    # Exit with appropriate code for CI/CD
    exit(0 if comparison["status"] == "PASS" else 1)

if __name__ == "__main__":
    main()
