from nemoguardrails import LLMRails, RailsConfig

# Define configuration
config = RailsConfig.from_content(
    colang_content="""
    # Define user intents
    define user ask about company
      "tell me about the company"
      "what does your company do"
      "company information"

    define user ask off topic
      "what's the weather"
      "tell me a joke"
      "who won the game"

    # Define bot responses
    define bot provide company info
      "We are a technology company focused on AI solutions.
       How can I help you with our products today?"

    define bot refuse off topic
      "I'm designed to help with questions about our products
       and services. Is there something specific I can help you with?"

    # Define dialog flows
    define flow company info
      user ask about company
      bot provide company info

    define flow handle off topic
      user ask off topic
      bot refuse off topic
    """,
    yaml_content="""
    models:
      - type: main
        engine: openai
        model: gpt-4o

    rails:
      input:
        flows:
          - check jailbreak
          - check toxicity
      output:
        flows:
          - check facts
          - check hallucination
    """
)

# Create rails instance

# Process a message
response = rails.generate(
    messages=[{
        "role": "user",
        "content": "Tell me about your company"
    }]
)

custom_guardrails.py 
"""Custom Guardrails Implementation

Build your own guardrail system with modular checks.
"""

import re
from typing import List, Dict, Tuple
from dataclasses import dataclass
from enum import Enum

class GuardrailAction(Enum):
    ALLOW = "allow"
    BLOCK = "block"
    MODIFY = "modify"
    FLAG = "flag"

@dataclass
class GuardrailResult:
    action: GuardrailAction
    message: str
    triggered_rules: List[str]
    modified_content: str = None

class Guardrail:
    """Base class for guardrails."""

    def __init__(self, name: str, enabled: bool = True):
        self.name = name
        self.enabled = enabled

    def check(self, content: str) -> Tuple[bool, str]:
        raise NotImplementedError

class PIIGuardrail(Guardrail):
    """Detect and redact personally identifiable information."""

    PATTERNS = {
        "email": r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b',
        "ssn": r'\b\d{3}-\d{2}-\d{4}\b',
        "phone": r'\b\d{3}[-.]?\d{3}[-.]?\d{4}\b',
        "credit_card": r'\b\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}\b'
    }

    def check(self, content: str) -> Tuple[bool, str]:
        found_pii = []
        redacted = content

        for pii_type, pattern in self.PATTERNS.items():
            matches = re.findall(pattern, content)
            if matches:
                found_pii.append(pii_type)
                redacted = re.sub(pattern, f"[REDACTED_{pii_type.upper()}]", redacted)

        if found_pii:
            return False, f"PII detected: {', '.join(found_pii)}"
        return True, "No PII detected"

class JailbreakGuardrail(Guardrail):
    """Detect jailbreak attempts."""

    PATTERNS = [
        r"ignore (all )?(previous |prior )?instructions",
        r"disregard (your )?(previous |prior )?instructions",
        r"you are now",
        r"act as if",
        r"pretend (you are|to be)",
        r"forget (everything|your rules)",
        r"bypass (your |the )?restrictions"
    ]

    def check(self, content: str) -> Tuple[bool, str]:
        content_lower = content.lower()

        for pattern in self.PATTERNS:
            if re.search(pattern, content_lower):
                return False, f"Potential jailbreak attempt detected"

        return True, "No jailbreak patterns detected"

class GuardrailPipeline:
    """Pipeline to run multiple guardrails."""

    def __init__(self):
        self.guardrails: List[Guardrail] = []

    def add(self, guardrail: Guardrail) -> 'GuardrailPipeline':
        self.guardrails.append(guardrail)
        return self

    def check(self, content: str) -> GuardrailResult:
        triggered = []
        action = GuardrailAction.ALLOW
        messages = []

        for guardrail in self.guardrails:
            if not guardrail.enabled:
                continue

            passed, message = guardrail.check(content)

            if not passed:
                triggered.append(guardrail.name)
                messages.append(message)
                action = GuardrailAction.BLOCK

        return GuardrailResult(
            action=action,
            message="; ".join(messages) if messages else "All checks passed",
            triggered_rules=triggered
        )

# Usage
pipeline = GuardrailPipeline()
pipeline.add(PIIGuardrail("pii_detector"))
pipeline.add(JailbreakGuardrail("jailbreak_detector"))

result = pipeline.check("My email is test@example.com")
print(f"Action: {result.action.value}")
print(f"Message: {result.message}")

"""Production Deployment Configuration

Complete setup for deploying agents to production.
"""

import os
from dataclasses import dataclass
from typing import Optional
import logging

@dataclass
class ProductionConfig:
    """Configuration for production deployment."""

    # API Configuration
    api_timeout: int = 30
    max_retries: int = 3
    retry_delay: float = 1.0

    # Rate Limiting
    requests_per_minute: int = 60
    tokens_per_minute: int = 100000

    # Cost Controls
    max_cost_per_request: float = 0.50
    daily_budget: float = 100.0

    # Safety Thresholds
    toxicity_threshold: float = 0.7
    jailbreak_confidence: float = 0.8

    # Monitoring
    enable_tracing: bool = True
    log_level: str = "INFO"
    alert_on_errors: bool = True

class ProductionAgent:
    """Production-ready agent wrapper with all safety features."""

    def __init__(self, config: ProductionConfig):
        self.config = config
        self.setup_logging()
        self.setup_monitoring()
        self.request_count = 0
        self.daily_cost = 0.0

    def setup_logging(self):
        logging.basicConfig(
            level=getattr(logging, self.config.log_level),
            format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        )
        self.logger = logging.getLogger("ProductionAgent")

    def setup_monitoring(self):
        if self.config.enable_tracing:
            os.environ["LANGCHAIN_TRACING_V2"] = "true"
            self.logger.info("Tracing enabled")

    def check_rate_limit(self) -> bool:
        """Check if request is within rate limits."""
        # Implement rate limiting logic
        return self.request_count < self.config.requests_per_minute

    def check_budget(self, estimated_cost: float) -> bool:
        """Check if request is within budget."""
        return (
            estimated_cost <= self.config.max_cost_per_request and
            self.daily_cost + estimated_cost <= self.config.daily_budget
        )

    async def process(self, message: str) -> dict:
        """Process a message with all safety checks."""

        # Rate limit check
        if not self.check_rate_limit():
            return {
                "error": "Rate limit exceeded",
                "retry_after": 60
            }

        # Budget check (estimate cost)
        estimated_cost = len(message) * 0.00001  # Simplified
        if not self.check_budget(estimated_cost):
            return {
                "error": "Budget limit reached",
                "daily_remaining": self.config.daily_budget - self.daily_cost
            }

        try:
            # Run guardrails on input
            guardrail_result = self.run_input_guardrails(message)
            if guardrail_result["blocked"]:
                return {
                    "error": "Input blocked by guardrails",
                    "reason": guardrail_result["reason"]
                }

            # Process message (your agent logic here)
            response = await self.agent.run(message)

            # Run guardrails on output
            output_result = self.run_output_guardrails(response)
            if output_result["modified"]:
                response = output_result["content"]

            # Update metrics
            self.request_count += 1
            self.daily_cost += estimated_cost

            return {
                "content": response,
                "cost": estimated_cost,
                "guardrails_triggered": guardrail_result.get("triggered", [])
            }

        except Exception as e:
            self.logger.error(f"Error processing request: {e}")

            if self.config.alert_on_errors:
                self.send_alert(str(e))

            return {
                "error": "An error occurred",
                "fallback": "I apologize, but I'm unable to process your request right now. Please try again later."
            }

    def run_input_guardrails(self, content: str) -> dict:
        # Implement input guardrails
        return {"blocked": False, "triggered": []}

    def run_output_guardrails(self, content: str) -> dict:
        # Implement output guardrails
        return {"modified": False, "content": content}

    def send_alert(self, message: str):
        # Implement alerting (PagerDuty, Slack, etc.)
        self.logger.critical(f"ALERT: {message}")

# Initialize for production
config = ProductionConfig(
    requests_per_minute=100,
    daily_budget=500.0,
    enable_tracing=True
)

agent = ProductionAgent(config)
monitoring_setup.py 
"""Monitoring and Alerting Setup

Set up comprehensive monitoring for production agents.
"""

import time
from dataclasses import dataclass, field
from typing import Dict, List, Callable
from datetime import datetime, timedelta
from collections import deque
import statistics

@dataclass
class MetricPoint:
    timestamp: datetime
    value: float
    labels: Dict[str, str] = field(default_factory=dict)

class MetricsCollector:
    """Collect and aggregate metrics for monitoring."""

    def __init__(self, window_minutes: int = 5):
        self.window = timedelta(minutes=window_minutes)
        self.metrics: Dict[str, deque] = {}

    def record(self, name: str, value: float, labels: Dict = None):
        """Record a metric value."""
        if name not in self.metrics:
            self.metrics[name] = deque(maxlen=1000)

        self.metrics[name].append(MetricPoint(
            timestamp=datetime.now(),
            value=value,
            labels=labels or {}
        ))

    def get_stats(self, name: str) -> Dict:
        """Get statistics for a metric."""
        if name not in self.metrics:
            return {}

        cutoff = datetime.now() - self.window
        recent = [p.value for p in self.metrics[name] if p.timestamp > cutoff]

        if not recent:
            return {}

        return {
            "count": len(recent),
            "mean": statistics.mean(recent),
            "p50": statistics.median(recent),
            "p95": statistics.quantiles(recent, n=20)[18] if len(recent) >= 20 else max(recent),
            "p99": statistics.quantiles(recent, n=100)[98] if len(recent) >= 100 else max(recent),
            "min": min(recent),
            "max": max(recent)
        }

class AlertRule:
    """Define alerting rules for metrics."""

    def __init__(
        self,
        name: str,
        metric: str,
        condition: Callable[[Dict], bool],
        message: str,
        severity: str = "warning"
    ):
        self.name = name
        self.metric = metric
        self.condition = condition
        self.message = message
        self.severity = severity
        self.last_triggered = None
        self.cooldown = timedelta(minutes=5)

    def check(self, stats: Dict) -> bool:
        """Check if alert should fire."""
        if not stats:
            return False

        if self.last_triggered and datetime.now() - self.last_triggered < self.cooldown:
            return False

        if self.condition(stats):
            self.last_triggered = datetime.now()
            return True

        return False

class MonitoringSystem:
    """Complete monitoring system for production agents."""

    def __init__(self):
        self.collector = MetricsCollector()
        self.alerts: List[AlertRule] = []
        self.alert_handlers: List[Callable] = []

    def add_alert(self, rule: AlertRule):
        self.alerts.append(rule)

    def add_alert_handler(self, handler: Callable):
        self.alert_handlers.append(handler)

    def record_request(self, latency_ms: float, tokens: int, cost: float, success: bool):
        """Record metrics for a request."""
        self.collector.record("latency_ms", latency_ms)
        self.collector.record("tokens", tokens)
        self.collector.record("cost", cost)
        self.collector.record("success", 1 if success else 0)

    def record_guardrail(self, guardrail_name: str, triggered: bool):
        """Record guardrail activation."""
        self.collector.record(
            f"guardrail_{guardrail_name}",
            1 if triggered else 0
        )

    def check_alerts(self):
        """Check all alert rules."""
        for rule in self.alerts:
            stats = self.collector.get_stats(rule.metric)
            if rule.check(stats):
                for handler in self.alert_handlers:
                    handler(rule, stats)

    def get_dashboard_data(self) -> Dict:
        """Get data for monitoring dashboard."""
        return {
            "latency": self.collector.get_stats("latency_ms"),
            "tokens": self.collector.get_stats("tokens"),
            "cost": self.collector.get_stats("cost"),
            "success_rate": self.collector.get_stats("success")
        }

# Set up monitoring
monitoring = MonitoringSystem()

# Add alert rules
monitoring.add_alert(AlertRule(
    name="high_latency",
    metric="latency_ms",
    condition=lambda s: s.get("p95", 0) > 5000,
    message="P95 latency exceeds 5 seconds",
    severity="warning"
))

monitoring.add_alert(AlertRule(
    name="error_rate",
    metric="success",
    condition=lambda s: s.get("mean", 1) < 0.95,
    message="Success rate dropped below 95%",
    severity="critical"
))

monitoring.add_alert(AlertRule(
    name="cost_spike",
    metric="cost",
    condition=lambda s: s.get("mean", 0) > 0.10,
    message="Average cost per request exceeds $0.10",
    severity="warning"
))

# Add alert handler
def slack_alert(rule: AlertRule, stats: Dict):
    print(f"[{rule.severity.upper()}] {rule.name}: {rule.message}")
    print(f"Current stats: {stats}")

monitoring.add_alert_handler(slack_alert)
